# zahlenraten_v2
# Zahlenratespiel mit Eingabesicherheit:
# - input() -> str
# - try/except fängt ValueError ab
# - innere Schleife wiederholt Eingabe bis gültige ganze Zahl

import random

zahl = random.randint(1, 100)
versuche = 0
geraten = 0

while zahl != geraten:
    while True:
        geraten = input("rate Zahl: ")
        try:
            geraten = int(geraten)
            break
        except ValueError:
            print("das ist keine ganze Zahl.")

    versuche = versuche + 1

    if geraten < zahl:
        print("Die Zahl ist zu klein.")
    elif geraten > zahl:
        print("Die Zahl ist zu groß.")
    else:
        # richtige Zahl -> äußere Schleife wird durch Bedingung beendet
        break

print("Du hast", versuche, "Versuche gebraucht.")

import random

runde = 0   # Zähler

def spiel_runde():
    zahl = random.randint(1, 100)
    versuche = 0
    geraten = 0

    while geraten != zahl:
        while True:
            geraten = input("rate die Zahl: ")
            try:
                geraten = int(geraten)
                break
            except ValueError:
                print("das ist keine Zahl!")

        versuche = versuche + 1

        if geraten > zahl:
            print("die Zahl ist zu groß!")
        elif geraten < zahl:
            print("die Zahl ist zu klein!")
        else:
            break

    versuche_zahlenraten.append(versuche)
    print("du hast", versuche, "Versuche gebraucht.")
    print()

    # Statistik ab der 2. Runde
    if runde >= 2:
        gesamt = sum(versuche_zahlenraten)
        anzahl = len(versuche_zahlenraten)
        durchschnitt = gesamt / anzahl

        print("Bis hier hast du folgende Versuche gebraucht:")
        print(versuche_zahlenraten)
        print("Beste Runde:", min(versuche_zahlenraten))
        print("Schlechteste Runde:", max(versuche_zahlenraten))
        print("Durchschnitt:", durchschnitt)
        print()
